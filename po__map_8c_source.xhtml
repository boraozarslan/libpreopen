43 struct po_map*
44 po_map_create(int capacity)
45 {
46  struct po_map *map;
47 
48  map = malloc(sizeof(struct po_map));
49  if (map == NULL) {
50  return (NULL);
51  }
52 
53  map->entries = calloc(sizeof(struct po_map_entry), capacity);
54  if (map->entries == NULL) {
55  free(map);
56  return (NULL);
57  }
58 
59  map->refcount = 1;
60  map->capacity = capacity;
61  map->length = 0;
62 
63  po_map_assertvalid(map);
64 
65  return (map);
66 }

68 struct po_map*
69 po_map_enlarge(struct po_map *map)
70 {
71  struct po_map_entry *enlarged;
72  enlarged = calloc(sizeof(struct po_map_entry), 2 * map->capacity);
73  if (enlarged == NULL) {
74  return (NULL);
75  }
76  memcpy(enlarged, map->entries, map->length * sizeof(*enlarged));
77  free(map->entries);
78  map->entries = enlarged;
79  map->capacity = 2 * map->capacity;
80  return map;
81 }

83 size_t
84 po_map_foreach(const struct po_map *map, po_map_iter_cb cb)
85 {
86  struct po_map_entry *entry;
87  size_t n;
88 
89  po_map_assertvalid(map);
90 
91  for (n = 0; n < map->length; n++) {
92  entry = map->entries + n;
93 
94  if (!cb(entry->name, entry->fd, entry->rights)) {
95  break;
96  }
97  }
98 
99  return (n);
100 }

102 void
103 po_map_release(struct po_map *map)
104 {
105  if (map == NULL) {
106  return;
107  }
108 
109  po_map_assertvalid(map);
110 
111  map->refcount -= 1;
112 
113  if (map->refcount == 0) {
114  free(map->entries);
115  free(map);
116  }
117 }
