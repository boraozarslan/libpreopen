45 /**
46  * @struct po_map
47  * @brief A mapping from paths to pre-opened directories.
48  *
49  * This type is opaque to clients, but it is reference-counted and can be
50  * thought of as containing a set (with no particular ordering guarantees)
51  * of path->dirfd mappings.
52  */

55 /**
56  * A callback that can be used to inpect the contents of a @ref po_map.
57  *
58  * This callback can be invoked by iteration code to expose elements of a

133 /**
134  * Pre-open a path and store it in a @ref po_map for later use.
135  *
136  * @param map the map to add the path->fd mapping to
137  * @param path the path to pre-open (which may be a directory if
138  * `O_DIRECTORY` is passed via @b flags)
139  * @param flags flags to pass to `open(2)` / `openat(2)`
140  * @param ... optional file mode, as accepted by `open(2)`
141  *
142  * @returns the file descriptor of the opened directory or -1 if
143  * @b path is not a directory or cannot be opened or if
144  * the @ref po_map cannot store the directory (e.g., resizing fails)
145  */
146 int po_preopen(struct po_map *map, const char *path, int flags, ...);

148 /**
149  * Find a directory whose path is a prefix of @b path and (on platforms that
150  * support Capsicum) that has the rights required by @b rights.
151  *
152  * @param map the map to look for a directory in

171 /**
172  * Pack a `struct po_map` into a shared memory segment.
173  *
174  * To inherit a `po_map` across the process execution boundary, it needs to be
175  * packed into an inheritable form such as a shared memory segment. This can
176  * then be unpacked in the child process for direct access.
177  *
178  * @param map the map to pack into shared memory
179  *
180  * @returns a file descriptor of a shared memory segment
181  * (or -1 on error)
182  */
183 int po_pack(struct po_map *map);
184 
185 /**
186  * Unpack a `struct po_map` from a file.

188  * Using the representation generated by `po_pack`, unpack a `po_map`
189  * and make it available for normal usage.
190  *
191  * @param fd a file containing a packed `po_map` representation
192  */
193 struct po_map* po_unpack(int fd);
